| **#sql实现**  set hive.auto.convert.join=false;  with temp as (  select  a.user\_id,a.regist\_dt,b.activity\_dt,  from\_unixtime(unix\_timestamp(a.regist\_dt,"yyyyMMdd"),"yyyy-MM-dd") as format\_regist\_dt,  datediff(from\_unixtime(unix\_timestamp(b.activity\_dt,"yyyyMMdd"),"yyyy-MM-dd"),from\_unixtime(unix\_timestamp(a.regist\_dt,"yyyyMMdd"),"yyyy-MM-dd")) as diff  from users a join user\_activity b on a.user\_id = b.user\_id  where b.activity\_dt >a.regist\_dt  )  select format\_regist\_dt,diff,count(\*) as cnt from temp where diff <=7 group by format\_regist\_dt,diff ;  **#结果**  +-------------------+-------+------+  | format\_regist\_dt | diff | cnt |  +-------------------+-------+------+  | 2024-06-01 | 1 | 3 |  | 2024-06-01 | 2 | 3 |  | 2024-06-01 | 3 | 3 |  | 2024-06-01 | 4 | 3 |  | 2024-06-01 | 7 | 2 |  | 2024-06-02 | 1 | 2 |  | 2024-06-02 | 2 | 2 |  | 2024-06-02 | 4 | 2 |  | 2024-06-02 | 7 | 2 |  | 2024-06-03 | 1 | 2 |  | 2024-06-03 | 3 | 1 |  | 2024-06-03 | 4 | 1 |  | 2024-06-03 | 7 | 1 |  | 2024-06-04 | 3 | 2 |  +-------------------+-------+------+ |
