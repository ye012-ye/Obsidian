| **val** session: SparkSession = SparkSession.*builder*().master(**"local"**).appName(**"test"**).getOrCreate() session.sparkContext.setLogLevel(**"Error"**)**val** df: DataFrame = session.read.option(**"header"**,**true**).csv(**"./data/MonitorData.txt"**)  df.createTempView(**"temp"**)  session.sql(  **"""**  **| select device\_id,state,dt,row\_number() over(partition by device\_id order by dt asc) as rk from temp**  **"""**.stripMargin).createTempView(**"temp1"**)   session.sql(  **"""**  **| select a.device\_id,a.state,a.dt**  **| from temp1 a join temp1 b on a.device\_id = b.device\_id and a.state != b.state and a.rk = b.rk -1**  **"""**.stripMargin).show() |
